167 //根据污染物数值查询计算配置的4个参数
168 func iaqConfig(pollutant int, node *j4g.JsonNode) (int, int, int, int) {
169     arrayNode := node.ArraysStruct
170     var cmin int = 0
171     var cmax int = 0
172     var imin int = 0
173     var imax int = 0
174     for i := 0; i < len(arrayNode); i++ {
175         cmin = int(arrayNode[i].ToJsonNode().GetNodeByName("cmin").ValueNumber)
176         cmax = int(arrayNode[i].ToJsonNode().GetNodeByName("cmax").ValueNumber)
177         imin = int(arrayNode[i].ToJsonNode().GetNodeByName("imin").ValueNumber)
178         imax = int(arrayNode[i].ToJsonNode().GetNodeByName("imax").ValueNumber)
179         if pollutant >= int(cmin) && pollutant < int(cmax) {
180             logger.Debug("计算公式【(imax-imin)/(cmax-cmin) *(pollint-cmin)+imin】【检测数值=", pollutant, "的计算方法", "(", imax, "-", imin, ")/(", cmax, "-", cmin, ")*(", pollutant, "-", cmin, ")+", cmin)
181             return cmin, cmax, imin, imax
182             break
183         }
184         if pollutant >= cmax && i == len(arrayNode)-1 {
185             logger.Debug("计算公式【(imax-imin)/(cmax-cmin) *(pollint-cmin)+imin】【检测数值=：", pollutant, "的计算方法", "(", imax, "-", imin, ")/(", cmax, "-", cmin, ")*(", pollutant, "-", cmin, ")+", cmin)
186 
187             return cmin, cmax, imin, imax
188         }
189     }
190     return 0, 0, 0, 0
191 }

193 //计算污染物iaq
194 func computeIAQ(pollutant int, cmin int, cmax int, imin int, imax int) int {
195     var temp float64 = 0
196     temp = float64(imax-imin) / float64(cmax-cmin)
197     result := temp*float64(pollutant-cmin) + float64(imin)
198     return int(result)
199 }

201 // 保存用户最好空气指数，和连续优良天数,同时更新 检测数据桶中的最新iaq信息
202 func setIAQToDB(sn string, iaq int, key string) {
203     b, err := couchbase.GetBucket(userProductUrl,
204         "default", "aicc-UserProductDB")
205     d, err := couchbase.GetBucket(couchDataUrl,

210     if err == nil {

212 
213         userProuctJson = string(upbyte)

215     } else {
216         logger.Error("保存iaq db 异常")
217     }
218     upNode, _ := j4g.LoadByString(userProuctJson)
219     mobileObj := upNode.GetNodeByName("mobile")
220     mobile := ""
221     if mobileObj != nil {
222         mobile = upNode.GetNodeByName("mobile").ValueString
223 
224     }
225 
226     if mobile != "" {
227         logger.Debug("设备【", sn, "】的【用户手机号】"+mobile)
228         userbyte, _ := b.GetRaw(mobile)
229         userJson = string(userbyte)
230         userNode, _ := j4g.LoadByString(userJson)
231         oldIaq := userNode.GetNodeByPath("iaq")
232         oldIaqCount := userNode.GetNodeByPath("iaqcount")
233 
234         if oldIaq == nil { //新增
235             logger.Debug("【", mobile, "】没有保存过iaq数据，准备新增")
236             iaqNode := new(j4g.JsonNode)
237             iaqCountNode := new(j4g.JsonNode)
238 
239             iaqNode.Name = "iaq"
240             iaqNode.SetValue(int(iaq))
241 
242             iaqCountNode.Name = "iaqcount"
243             iaqCountNode.SetValue(1)
244 
245             userNode.AddNode(iaqNode)
246             userNode.AddNode(iaqCountNode)
247 
248             b.SetRaw(mobile, 0, []byte(userNode.ToCouchDBString()))
249 
250             logger.Debug("...................用户【", mobile, "】新增iaq 信息已经保存")
251         } else {
252             // 更新iaq 和iaqcount
253             if iaq <= int(oldIaq.ValueNumber) {
254                 //更新最小的iaq
255                 iaqNode := new(j4g.JsonNode)
256                 iaqCountNode := new(j4g.JsonNode)
257 
258                 iaqNode.Name = "iaq"
259                 iaqNode.SetValue(int(iaq))
260 
261                 iaqCountNode.Name = "iaqcount"
262 
263                 userNode.AddNode(iaqNode)
264 
265                 if iaq < 100 {
266                     iaqCountNode.SetValue(int(oldIaqCount.ValueNumber) + 1)
267                     userNode.AddNode(iaqCountNode)
268                 } else {
269                     iaqCountNode.SetValue(0)
270                     userNode.AddNode(iaqCountNode)
271                 }
272                 b.SetRaw(mobile, 0, []byte(userNode.ToCouchDBString()))
273                 logger.Info("更新用户室内最好IAQ【", mobile, "】oldiaq=", int(oldIaq.ValueNumber), " newiaq=", iaq, "oldIaqCount=", int(oldIaqCount.ValueNumber), "new iaqCount=", int(oldIaqCount.ValueNumber)+1)
274 
275             } else {
276                 //空气质量变差，从新计数
277                 iaqCountNode := new(j4g.JsonNode)
278                 iaqCountNode.Name = "iaqcount"
279                 iaqCountNode.SetValue(0)
280                 if iaq < 100 {
281                     iaqCountNode.SetValue(int(oldIaqCount.ValueNumber) + 1)
282                     userNode.AddNode(iaqCountNode)
283                 } else {
284                     iaqCountNode.SetValue(0)
285                     userNode.AddNode(iaqCountNode)
286                 }
287                 b.SetRaw(mobile, 0, []byte(userNode.ToCouchDBString()))
288                 logger.Debug("【 用户 ", mobile, "空气质量变差重新开始累计】.......................")
289             }
290         }
291 
292     }
293     //把iaq更新到 couchdb中
294     dataByte, _ := d.GetRaw(key)
295     dataJson := string(dataByte)
296     dataNode, _ := j4g.LoadByString(dataJson)
297     iaqNode := new(j4g.JsonNode)
298     iaqNode.Name = "iaq"
299     iaqNode.SetValue(int(iaq))
300     dataNode.AddNode(iaqNode)
301     logger.Info("【更新数据桶中iaq", key, "】-->", dataNode.GetNodeByPath("iaq").ValueNumber)
302 
303     d.SetRaw(key, 0, []byte(dataNode.ToCouchDBString()))
304     defer func() {
305         if er := recover(); er != nil {
306             logger.Error(".............发生严重错误............................")
307             logger.Error(er)
308             logger.Error(".............发生严重错误............................")
309 
310         }
311         b.Close()
312         d.Close()
313     }()
314 }
